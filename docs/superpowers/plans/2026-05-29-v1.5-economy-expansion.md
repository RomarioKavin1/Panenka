```solidity
// SPDX-License-Identifier: MIT
pragma solidity 0.8.24;

import {Test} from "forge-std/Test.sol";
import {PackSale} from "../src/PackSale.sol";
import {CardNFT} from "../src/CardNFT.sol";
import {MockUSDC} from "../src/mocks/MockUSDC.sol";

contract PackSaleDiamondTest is Test {
    PackSale pack;
    CardNFT card;
    MockUSDC usdc;
    address owner = address(0xA);
    address buyer = address(0xB);
    address treasury = address(0xC);

    uint8 constant DIAMOND = 3;

    function setUp() public {
        vm.startPrank(owner);
        usdc = new MockUSDC();
        card = new CardNFT();
        pack = new PackSale(address(card), address(usdc), treasury);
        card.setMinter(address(pack), true);

        // Configure 1 player; set stats for all tiers
        bytes32[] memory pool = new bytes32[](1);
        pool[0] = keccak256("PLAYER-1");
        pack.setPlayerPool(pool);
        for (uint8 t = 0; t < 4; t++) {
            CardNFT.Stats memory s = CardNFT.Stats(50, 50, 50, 50, 50);
            card.setPlayerStats(pool[0], t, s);
        }

        // Diamond price + cumulative weights — Diamond always pulls Unique (tier 3)
        pack.setPackPrice(DIAMOND, 100e6); // 100 USDC
        // tierCum for Diamond: [0, 0, 0, 10000] — pure Unique
        pack.setTierCum(DIAMOND, [uint16(0), 0, 0, 10000]);
        vm.stopPrank();
    }

    function test_DiamondPackMintsAtLeastOneUnique() public {
        usdc.faucet(1000e6);
        usdc.transfer(buyer, 1000e6);
        vm.startPrank(buyer);
        usdc.approve(address(pack), 100e6);
        uint256 commitId = pack.buy(DIAMOND);
        vm.roll(block.number + 16);
        pack.reveal(commitId);
        vm.stopPrank();

        // Buyer should own 5 cards, at least one Unique (tier 3)
        uint256 uniques = 0;
        for (uint256 i = 1; i <= 5; i++) {
            if (card.ownerOf(i) == buyer && card.tierOf(i) == 3) uniques++;
        }
        assertGe(uniques, 1, "expected >= 1 Unique in Diamond pack");
    }

    function test_DiamondPackRespectsUniqueSupplyCap() public {
        usdc.faucet(1000e6);
        usdc.transfer(buyer, 1000e6);
        vm.startPrank(buyer);
        usdc.approve(address(pack), 1000e6);

        // Buy + reveal first Diamond — mints Unique #1
        uint256 c1 = pack.buy(DIAMOND);
        vm.roll(block.number + 16);
        pack.reveal(c1);

        // Buy + reveal second Diamond — Unique pool exhausted, downgrades to SR
        uint256 c2 = pack.buy(DIAMOND);
        vm.roll(block.number + 16);
        pack.reveal(c2);
        vm.stopPrank();

        // Check no two Unique cards for the same player
        uint256 uniqueCount = 0;
        for (uint256 i = 1; i <= 10; i++) {
            try card.tierOf(i) returns (uint8 t) {
                if (t == 3) uniqueCount++;
            } catch { break; }
        }
        assertEq(uniqueCount, 1, "supply cap of 1 Unique per player must hold");
    }
}
```

**Files:**

- [ ] **Step 1:** Implement the contract:

```solidity
// SPDX-License-Identifier: MIT
pragma solidity 0.8.24;

import {IERC20} from "@openzeppelin/contracts/token/ERC20/IERC20.sol";
import {IERC721} from "@openzeppelin/contracts/token/ERC721/IERC721.sol";
import {SafeERC20} from "@openzeppelin/contracts/token/ERC20/utils/SafeERC20.sol";
import {Ownable} from "@openzeppelin/contracts/access/Ownable.sol";

contract UniqueAuction is Ownable {
    using SafeERC20 for IERC20;

    struct AuctionInfo {
        address seller;
        uint256 reserve;
        uint256 highestBid;
        address highestBidder;
        uint64 endsAt;
        address federation;
        uint16 federationBps;   // share to federation, e.g. 3000 = 30%
        bool settled;
    }

    uint256 public constant ANTI_SNIPE_EXTENSION = 10 minutes;
    uint256 public constant ANTI_SNIPE_WINDOW = 10 minutes; // bid within last 10 min triggers extension

    IERC721 public immutable card;
    IERC20 public immutable usdc;
    address public treasury;

    mapping(uint256 => AuctionInfo) public auctions;

    event Started(uint256 indexed tokenId, address indexed seller, uint256 reserve, uint64 endsAt);
    event BidPlaced(uint256 indexed tokenId, address indexed bidder, uint256 amount, uint64 newEndsAt);
    event Settled(uint256 indexed tokenId, address winner, uint256 amount);
    event Rescued(uint256 indexed tokenId, address seller);

    constructor(address cardAddr, address usdcAddr, address treasuryAddr) Ownable(msg.sender) {
        card = IERC721(cardAddr);
        usdc = IERC20(usdcAddr);
        treasury = treasuryAddr;
    }

    function setTreasury(address t) external onlyOwner { treasury = t; }

    function start(
        uint256 tokenId,
        uint256 reserve,
        uint256 duration,
        address federation,
        uint16 federationBps
    ) external returns (uint64 endsAt) {
        require(auctions[tokenId].seller == address(0), "active");
        require(reserve > 0, "reserve");
        require(federationBps <= 10000, "bps");

        card.transferFrom(msg.sender, address(this), tokenId);

        endsAt = uint64(block.timestamp + duration);
        auctions[tokenId] = AuctionInfo({
            seller: msg.sender,
            reserve: reserve,
            highestBid: 0,
            highestBidder: address(0),
            endsAt: endsAt,
            federation: federation,
            federationBps: federationBps,
            settled: false
        });
        emit Started(tokenId, msg.sender, reserve, endsAt);
    }

    function bid(uint256 tokenId, uint256 amount) external {
        AuctionInfo storage a = auctions[tokenId];
        require(a.seller != address(0), "no auction");
        require(block.timestamp < a.endsAt, "ended");
        require(amount >= a.reserve, "below reserve");
        require(amount > a.highestBid, "must outbid");

        usdc.safeTransferFrom(msg.sender, address(this), amount);

        // Refund previous bidder
        if (a.highestBidder != address(0)) {
            usdc.safeTransfer(a.highestBidder, a.highestBid);
        }

        a.highestBid = amount;
        a.highestBidder = msg.sender;

        // Anti-snipe: bids in the last ANTI_SNIPE_WINDOW extend by ANTI_SNIPE_EXTENSION
        if (a.endsAt - block.timestamp < ANTI_SNIPE_WINDOW) {
            a.endsAt = uint64(a.endsAt + ANTI_SNIPE_EXTENSION);
        }

        emit BidPlaced(tokenId, msg.sender, amount, a.endsAt);
    }

    function settle(uint256 tokenId) external {
        AuctionInfo storage a = auctions[tokenId];
        require(a.seller != address(0), "no auction");
        require(block.timestamp >= a.endsAt, "not ended");
        require(!a.settled, "settled");
        require(a.highestBidder != address(0), "no bids");

        a.settled = true;

        // Split proceeds: federationBps to federation, rest to treasury
        uint256 federationCut = (a.highestBid * a.federationBps) / 10000;
        uint256 treasuryCut = a.highestBid - federationCut;
        if (federationCut > 0 && a.federation != address(0)) {
            usdc.safeTransfer(a.federation, federationCut);
        }
        usdc.safeTransfer(treasury, treasuryCut);

        card.transferFrom(address(this), a.highestBidder, tokenId);
        emit Settled(tokenId, a.highestBidder, a.highestBid);
    }

    function rescue(uint256 tokenId) external {
        AuctionInfo storage a = auctions[tokenId];
        require(a.seller == msg.sender, "not seller");
        require(block.timestamp >= a.endsAt, "not ended");
        require(a.highestBidder == address(0), "had bids");
        require(!a.settled, "settled");
        a.settled = true;
        card.transferFrom(address(this), msg.sender, tokenId);
        emit Rescued(tokenId, msg.sender);
    }

    function auctionOf(uint256 tokenId) external view returns (AuctionInfo memory) {
        return auctions[tokenId];
    }
}
```

```solidity
import {MerkleProof} from "@openzeppelin/contracts/utils/cryptography/MerkleProof.sol";
import {Ownable} from "@openzeppelin/contracts/access/Ownable.sol";

interface IChipMintable {
    function mint(address to, uint256 id, uint256 amount) external;
}

contract ChipDropMinter is Ownable {
    IChipMintable public immutable chip;
    address public oracle;

    mapping(uint256 => bytes32) public dropRoots;       // dropId → merkle root
    mapping(uint256 => mapping(address => bool)) public claimed;

    event OracleSet(address indexed oracle);
    event DropRootSet(uint256 indexed dropId, bytes32 root);
    event Claimed(uint256 indexed dropId, address indexed account, uint8 chipId, uint16 amount);

    constructor(address chipAddr) Ownable(msg.sender) {
        chip = IChipMintable(chipAddr);
    }

    function setOracle(address o) external onlyOwner {
        oracle = o;
        emit OracleSet(o);
    }

    function setDropRoot(uint256 dropId, bytes32 root) external {
        require(msg.sender == oracle, "not oracle");
        require(dropRoots[dropId] == bytes32(0), "already set");
        dropRoots[dropId] = root;
        emit DropRootSet(dropId, root);
    }

    function claim(uint256 dropId, uint8 chipId, uint16 amount, bytes32[] calldata proof) external {
        bytes32 root = dropRoots[dropId];
        require(root != bytes32(0), "no drop");
        require(!claimed[dropId][msg.sender], "claimed");

        bytes32 leaf = keccak256(abi.encodePacked(msg.sender, chipId, amount));
        require(MerkleProof.verify(proof, root, leaf), "bad proof");

        claimed[dropId][msg.sender] = true;
        chip.mint(msg.sender, chipId, amount);
        emit Claimed(dropId, msg.sender, chipId, amount);
    }
}
```

```typescript
// services/oracle/chipDrops.ts
import { keccak256, encodePacked } from "viem";
import { buildMerkleTree } from "@/lib/business/merkle";
import type { LineupScoreResult } from "./score";

export interface ChipDropEntry {
  wallet: `0x${string}`;
  chipId: number;
  amount: number;
}

export function chipDropLeaf(wallet: string, chipId: number, amount: number): `0x${string}` {
  return keccak256(encodePacked(["address", "uint8", "uint16"], [wallet as `0x${string}`, chipId, amount]));
}

/**
 * Compute earned drops for a matchday.
 *
 * v1.5 award rules:
 *  - Top 1% of paid-contest winners → 1x Wildcard (chipId 2)
 *  - Hat-trick performer's lineup owner (top 5 with that player) → 1x Triple Captain (chipId 0)
 */
export function computeChipDrops(
  scoresByWallet: Map<string, number>,
  hatTrickWallets: string[],
): ChipDropEntry[] {
  const entries: ChipDropEntry[] = [];

  // Top 1% by score get a Wildcard
  const sorted = Array.from(scoresByWallet.entries()).sort((a, b) => b[1] - a[1]);
  const topCount = Math.max(1, Math.floor(sorted.length * 0.01));
  for (const [wallet] of sorted.slice(0, topCount)) {
    entries.push({ wallet: wallet as `0x${string}`, chipId: 2, amount: 1 });
  }

  // Hat-trick rewards
  for (const w of hatTrickWallets) {
    entries.push({ wallet: w as `0x${string}`, chipId: 0, amount: 1 });
  }

  return entries;
}

export function buildChipDropRoot(entries: ChipDropEntry[]): `0x${string}` {
  if (entries.length === 0) return ("0x" + "00".repeat(32)) as `0x${string}`;
  const leaves = entries.map((e) => chipDropLeaf(e.wallet, e.chipId, e.amount));
  return buildMerkleTree(leaves);
}
```

## Risks captured

- Diamond pack supply depends on Unique cards minting via the same `tierSupplyCap` enforcement; if all Uniques are auctioned (Phase B) before Diamond packs are bought, the `_mintWithDowngrade` path activates and Diamond pulls degrade to SR. This is by design but should be communicated to users buying Diamond packs.
- Auction federation/charity split (`federationBps`) is per-auction owner-configurable; could be set to 0% for testnet demo. Production launches should enforce a minimum federation share via owner-only governance.
- Chip-drop awards via top 1% are sensitive to participation level — early matchdays with few entrants will mint disproportionately. v1.5 should ship a hard cap (e.g., max 50 chips per drop_id) — add a config in `ChipDropMinter.setMintCap` similar to `ChipNFT.setMintCap` in v1.
- Private leagues are entirely off-chain; if Supabase is unavailable, leagues don't function. Public contests remain on-chain regardless.
